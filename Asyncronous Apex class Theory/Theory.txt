public DatabaseQueryLocator start(Database.BatchableContext bc){
//u will extract the data through query(which records to be considered for the job)
return DatabaseQueryLocator.getQueryLocator('Select id from Account');

}
public void execute(Database.BatchableContext bc,List<SObject>){
//write logic with the batch of records sent from start()

}
public void finish(DatabaseBatchableContext bc){
// after processing of records tasks (includes chaining of jobs or queue but not future method)
DataBase.execute(new SecondBatchJob())
}

}

Call: Database.executeBatch(new BatchJobDemo,batchsize); //batchsize=> Integer

Minimum batch size - 1
Max- 2000
